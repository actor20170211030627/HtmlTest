<script type="text/javascript">
 $(function(){

  //在给定的祖先元素下匹配所有的后代元素(儿子,孙子...)
  $("#btn1").click(function(){
   $("body div").css("background-color","gold");//body内所有叫div的(儿子/孙子)...
  });

  //在给定的父元素下匹配所有的子元素(儿子)
  $("#btn2").click(function(){
   $("body>div").css("background-color","gold");//body内所有的div(儿子)
  });

  //匹配所有紧接在 prev 元素后的 next 元素(下一个同层级元素)
  $("#btn3").click(function(){
   $("#two+div").css("background-color","gold");//id为two的下一个同层级叫div的元素(同桌)
  });

  //匹配 prev 元素之后的所有 siblings 元素(同一种元素的兄弟)
  $("#btn4").click(function(){
   $("#one~div").css("background-color","gold");//id为one的所有同层级叫div的元素(兄弟)
  });
 });
</script>
